```python
# agent/response_cache.py
class ResponseCache:
    """LLM 响应缓存"""
    
    def __init__(self, max_size=1000, ttl_seconds=3600):
        self.cache = LRU(maxsize=max_size)
        self.ttl = ttl_seconds
    
    def get(self, prompt_hash):
        """获取缓存的响应"""
        if prompt_hash in self.cache:
            entry = self.cache[prompt_hash]
            if time.time() - entry['timestamp'] &lt; self.ttl:
                return entry['response']
        return None
    
    def put(self, prompt_hash, response):
        """缓存响应"""
        self.cache[prompt_hash] = {
            'response': response,
            'timestamp': time.time()
        }
```

**缓存策略**：

# memory/intelligent_summarizer.py
class IntelligentSummarizer:
    """智能摘要器"""
    
    def __init__(self):
        self.history = []
        self.compression_threshold = 0.8  # 80%
        self.last_compression = time.time()
    
    def should_compress(self, current_size, limit):
        """智能判断是否需要压缩"""
        ratio = current_size / limit
        
        # 1. 超过阈值必须压缩
        if ratio &gt;= self.compression_threshold:
            return True
        
        # 2. 内存压力大时提前压缩
        if self._is_memory_high():
            return ratio &gt; 0.6
        
        # 3. 避免频繁压缩
        if time.time() - self.last_compression &lt; 60:
            return False
        
        return False
    
    def _is_memory_high(self):
        import psutil
        mem = psutil.virtual_memory()
        return mem.percent &gt; 85

# utils/object_pool.py
class ObjectPool:
    """对象池"""
    
    def __init__(self, object_factory, max_size=50):
        self.factory = object_factory
        self.pool = []
        self.max_size = max_size
        self.lock = threading.Lock()
    
    def acquire(self):
        with self.lock:
            if self.pool:
                return self.pool.pop()
            return self.factory()
    
    def release(self, obj):
        with self.lock:
            if len(self.pool) &lt; self.max_size:
                self.pool.append(obj)
```

#### 5.2.3 分片存储

# agent/memory/sharded_store.py
class ShardedVectorStore:
    """分片向量存储"""
    
    def __init__(self, n_shards=4):
        self.n_shards = n_shards
        self.shards = [VectorStore(f"shard_{i}") for i in range(n_shards)]
    
    def _get_shard(self, key):
        """一致性哈希选择分片"""
        return hash(key) % self.n_shards
    
    def add(self, content, metadata=None):
        shard_idx = self._get_shard(content)
        return self.shards[shard_idx].add(content, metadata)
    
    def search(self, query, top_k=5):
        """并行搜索所有分片"""
        from concurrent.futures import ThreadPoolExecutor
        
        with ThreadPoolExecutor(max_workers=self.n_shards) as executor:
            futures = [
                executor.submit(shard.search, query, top_k)
                for shard in self.shards
            ]
            results = []
            for future in futures:
                results.extend(future.result())
        
        # 合并结果
        return self._merge_results(results, top_k)
```

```python
# agent/safety_guard.py (enhanced)
class IntelligentCircuitBreaker:
    """智能熔断器"""
    
    def __init__(self, name, failure_threshold=5, recovery_timeout=30):
        self.name = name
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        
        self.state = 'closed'
        self.failure_count = 0
        self.last_failure_time = 0
        
        # 自适应阈值
        self.adaptive_threshold = True
        self.historical_failures = []
    
    def call(self, func, *args, fallback=None, **kwargs):
        """调用函数，带熔断保护"""
        if self.state == 'open':
            if time.time() - self.last_failure_time &gt; self.recovery_timeout:
                self.state = 'half-open'
            else:
                if fallback:
                    return fallback()
                raise CircuitBreakerOpen(self.name)
        
        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure()
            if fallback:
                return fallback()
            raise
    
    def _on_success(self):
        if self.state == 'half-open':
            self.state = 'closed'
            self.failure_count = 0
        elif self.state == 'closed':
            self.failure_count = max(0, self.failure_count - 1)
    
    def _on_failure(self):
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        self.historical_failures.append(time.time())
        if len(self.historical_failures) &gt; 100:
            self.historical_failures.pop(0)
        
        threshold = self._compute_adaptive_threshold()
        if self.failure_count &gt;= threshold:
            self.state = 'open'
    
    def _compute_adaptive_threshold(self):
        if not self.adaptive_threshold or len(self.historical_failures) &lt; 10:
            return self.failure_threshold
        
        # 计算历史失败频率，动态调整阈值
        now = time.time()
        recent_failures = sum(1 for t in self.historical_failures if now - t &lt; 60)
        
        if recent_failures &gt; 20:
            return max(3, self.failure_threshold - 2)
        elif recent_failures &lt; 5:
            return min(10, self.failure_threshold + 1)
        
        return self.failure_threshold
```

            },

            },

                'enable_llm': True,

                'enable_lifetrace': False,
                'enable_vector_search': False
            },
            'heavy': {
                'enable_llm': False,  # 使用本地响应
                'enable_reflection': False,
                'enable_lifetrace': False,
                'enable_vector_search': False
            }
        }
        
        self.current_level = 'normal'
    
    def evaluate(self, health_metrics):
        """评估并决定降级级别"""
        cpu = health_metrics.get('cpu_percent', 0)
        memory = health_metrics.get('memory_percent', 0)
        latency = health_metrics.get('avg_latency_ms', 0)
        
        if cpu &gt; 95 or memory &gt; 95 or latency &gt; 5000:
            return 'heavy'
        elif cpu &gt; 80 or memory &gt; 80 or latency &gt; 2000:
            return 'medium'
        elif cpu &gt; 60 or memory &gt; 60 or latency &gt; 1000:
            return 'light'
        
        return 'normal'
    
    def apply(self, level):
        """应用降级级别"""
        self.current_level = level
        logger.info(f"降级到: {level}")
