```python
    # ... 原有初始化代码保持不变 ...

    planning_config = config.get("planning", {}) if config else {}

    # 创建工具注册表
    self._planning_tools = ToolRegistry()

    self._planner = PlanningCore(

    self._react_loop = ReActLoop(
        planner=self._planner,

        max_iterations=planning_config.get("max_iterations", 10)
```

#### 3.5.3 注册规划工具(新方法)

```python
def _register_planning_tools(self):
    """注册规划引擎可用的工具"""

    # 注册内置工具
    self._planning_tools.register(
        "check_health",
        self.check_health,
        {
            "description": "检查云枢的身体状态",
            "params": {}
        }
    )

    self._planning_tools.register(
        "get_status",
        lambda: self.get_status(),
        {
            "description": "获取云枢的完整状态",
            "params": {}
        }
    )

    self._planning_tools.register(
        "search_memory",
        lambda query: self._memory.query_logs(search=query, limit=10),
        {
            "description": "搜索云枢的记忆",
            "params": {"query": "搜索关键词"}
        }
    )

    self._planning_tools.register(
        "get_sensor_summary",
        lambda: self.body.get_sensor_summary(),
        {
            "description": "获取所有传感器状态摘要",
            "params": {}
        }
    )

    # 注册LLM对话工具
    self._planning_tools.register(
        "llm_chat",
        self._planning_llm_chat,
        {
            "description": "与用户进行对话",
            "params": {"response": "回复内容"}
        }
    )

    logger.info(f"已注册 {len(self._planning_tools.list_tools())} 个规划工具")

async def _planning_llm_chat(self, response: str = None, **kwargs) -> str:
    """规划引擎使用的LLM对话接口"""
    if response:
        return response
    return "请提供回复内容"
```

#### 3.5.4 扩展处理流程(修改 chat 或 process_user_input)

```python
def chat(self, user_input: str) -> str:
    """与云枢对话 - 增强版,支持规划"""

    if not self._running:
        return "我还没有被唤醒。请先调用 start() 让我醒来。"

    self._interaction_count += 1

    # 评估任务复杂度
    if self._needs_planning(user_input):
        # 复杂任务: 使用规划引擎
        return self._planning_chat(user_input)
    else:
        # 简单任务: 直接处理
        return self._direct_chat(user_input)

def _needs_planning(self, task: str) -> bool:
    """判断任务是否需要规划"""
    complex_indicators = [
        "帮我完成", "帮我创建", "帮我分析",
        "帮我构建", "流程", "系统",
        "第一步", "第二步"
    ]

    score = sum(1 for indicator in complex_indicators
                if indicator in task)

    # 超过阈值或包含多个动作词
    action_verbs = ["检查", "分析", "创建", "生成", "整理", "监控"]
    action_count = sum(1 for verb in action_verbs if verb in task)

    return score >= 1 or action_count >= 2

async def _planning_chat(self, user_input: str) -> str:
    """规划模式处理复杂任务"""
    # 构建上下文
    context = {
        "body_status": self._build_body_status(self.check_health()),
        "behavior_mode": self._current_mode.value,
    }

    try:
        # 执行ReAct循环
        result = await self._react_loop.run(user_input, context)

        if result.success:
            response = str(result.result)
        else:
            response = f"我遇到了一些问题: {result.error}"

        # 添加执行信息(可选)
        if result.iterations > 1:
            response += f"\n\n(经过 {result.iterations} 步处理)"

    except Exception as e:
        logger.error(f"规划处理失败: {e}")
        response = f"抱歉,处理这个任务时遇到了问题: {e}"

    # 记录到记忆
    self._memory.add_message("user", user_input)
    self._memory.add_message("assistant", response)

    return response

def _direct_chat(self, user_input: str) -> str:
    """直接模式处理简单任务 - 原有逻辑"""
    # 复用原有的 _process_user_input 或类似逻辑
    readings = self.check_health()
    body_status = self._build_body_status(readings)

    return self._call_llm(user_input, body_status)
```

## 六、故障排查

### 6.1 常见问题

| 问题 | 可能原因 | 解决方案 |
|------|----------|----------|
| 规划引擎未激活 | 配置未更新 | 检查 `planning.enabled` 配置 |
| 工具调用失败 | 工具未注册 | 调用 `_register_planning_tools()` |
| 循环执行 | 任务无法完成 | 检查 `max_iterations` 设置 |
| LLM调用失败 | API配置错误 | 检查 `LLM_PROVIDER` 等环境变量 |
